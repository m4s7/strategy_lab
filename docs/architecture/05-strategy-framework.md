# Strategy Lab Technical Architecture - Strategy Framework

## Strategy Framework Overview

The strategy framework provides a flexible, extensible system for developing and testing trading strategies. It uses a plugin architecture that allows strategies to be developed, tested, and swapped without modifying core infrastructure.

## Strategy Interface Design

### Base Strategy Abstract Class

```python
from abc import ABC, abstractmethod
from typing import Dict, Any, Optional
from dataclasses import dataclass

@dataclass
class MarketData:
    """Market data snapshot provided to strategies"""
    timestamp: int
    bid: float
    ask: float
    last: float
    volume: int
    book_state: Optional[OrderBookState] = None

@dataclass
class Signal:
    """Trading signal generated by strategy"""
    action: str  # 'BUY', 'SELL', 'HOLD'
    quantity: int
    price: Optional[float] = None
    order_type: str = 'MARKET'  # 'MARKET', 'LIMIT'
    metadata: Dict[str, Any] = None

class BaseStrategy(ABC):
    """Base class for all trading strategies"""
    
    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.position = 0
        self.state = {}
        self._validate_config()
    
    @abstractmethod
    def generate_signal(self, market_data: MarketData) -> Signal:
        """Generate trading signal based on market data"""
        pass
    
    @abstractmethod
    def get_parameters(self) -> Dict[str, Any]:
        """Return strategy parameters for optimization"""
        pass
    
    def update_parameters(self, params: Dict[str, Any]) -> None:
        """Update strategy parameters"""
        self._validate_parameters(params)
        self.config.update(params)
    
    def reset_state(self) -> None:
        """Reset strategy state for new backtest"""
        self.position = 0
        self.state = {}
        self.on_reset()
    
    def on_reset(self) -> None:
        """Override for custom reset logic"""
        pass
    
    def on_fill(self, fill_price: float, fill_quantity: int) -> None:
        """Handle order fill notifications"""
        self.position += fill_quantity
    
    def _validate_config(self) -> None:
        """Validate strategy configuration"""
        required = self.get_required_config()
        for key in required:
            if key not in self.config:
                raise ValueError(f"Missing required config: {key}")
    
    def get_required_config(self) -> List[str]:
        """Override to specify required configuration keys"""
        return []
```

## Strategy Registry System

### Dynamic Strategy Registration

```python
class StrategyRegistry:
    """Registry for managing available strategies"""
    
    def __init__(self):
        self._strategies: Dict[str, Type[BaseStrategy]] = {}
        self._metadata: Dict[str, StrategyMetadata] = {}
    
    def register(self, 
                 name: str, 
                 strategy_class: Type[BaseStrategy],
                 metadata: Optional[StrategyMetadata] = None) -> None:
        """Register a strategy class"""
        if name in self._strategies:
            raise ValueError(f"Strategy '{name}' already registered")
            
        self._strategies[name] = strategy_class
        self._metadata[name] = metadata or StrategyMetadata(name=name)
    
    def get_strategy(self, name: str, config: Dict[str, Any]) -> BaseStrategy:
        """Instantiate strategy by name"""
        if name not in self._strategies:
            raise ValueError(f"Unknown strategy: {name}")
        
        return self._strategies[name](config)
    
    def list_strategies(self) -> List[StrategyInfo]:
        """List all registered strategies with metadata"""
        return [
            StrategyInfo(
                name=name,
                class_name=cls.__name__,
                metadata=self._metadata.get(name)
            )
            for name, cls in self._strategies.items()
        ]
    
    def get_default_config(self, name: str) -> Dict[str, Any]:
        """Get default configuration for strategy"""
        if name not in self._strategies:
            raise ValueError(f"Unknown strategy: {name}")
        
        strategy_class = self._strategies[name]
        return strategy_class.get_default_config()

# Global registry instance
strategy_registry = StrategyRegistry()

# Decorator for automatic registration
def register_strategy(name: str, **metadata):
    """Decorator to register strategy classes"""
    def decorator(cls):
        strategy_registry.register(
            name, 
            cls, 
            StrategyMetadata(name=name, **metadata)
        )
        return cls
    return decorator
```

## Example Strategy Implementations

### Order Book Imbalance Strategy

```python
@register_strategy(
    "orderbook_imbalance",
    description="Trades based on order book imbalance",
    requires_l2_data=True
)
class OrderBookImbalanceStrategy(BaseStrategy):
    """Strategy based on order book imbalance"""
    
    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)
        self.imbalance_threshold = config.get('imbalance_threshold', 0.6)
        self.min_spread = config.get('min_spread', 0.25)
        self.position_size = config.get('position_size', 1)
        self.max_position = config.get('max_position', 3)
        
        # Strategy state
        self.imbalance_history = deque(maxlen=100)
        
    def generate_signal(self, market_data: MarketData) -> Signal:
        """Generate signal based on order book imbalance"""
        if not market_data.book_state:
            return Signal('HOLD', 0)
        
        # Calculate current metrics
        imbalance = market_data.book_state.get_imbalance()
        spread = market_data.ask - market_data.bid
        
        # Track imbalance history
        self.imbalance_history.append(imbalance)
        
        # Check spread requirement
        if spread < self.min_spread:
            return Signal('HOLD', 0, metadata={'reason': 'spread_too_narrow'})
        
        # Position limits
        if abs(self.position) >= self.max_position:
            return Signal('HOLD', 0, metadata={'reason': 'position_limit'})
        
        # Generate signals based on imbalance
        if imbalance > self.imbalance_threshold:
            return Signal(
                'BUY', 
                self.position_size,
                price=market_data.ask,  # Aggressive fill
                order_type='LIMIT',
                metadata={'imbalance': imbalance, 'spread': spread}
            )
        elif imbalance < -self.imbalance_threshold:
            return Signal(
                'SELL', 
                self.position_size,
                price=market_data.bid,  # Aggressive fill
                order_type='LIMIT',
                metadata={'imbalance': imbalance, 'spread': spread}
            )
        
        return Signal('HOLD', 0)
    
    def get_parameters(self) -> Dict[str, Any]:
        """Define optimizable parameters"""
        return {
            'imbalance_threshold': {
                'min': 0.3,
                'max': 0.8,
                'step': 0.05,
                'default': 0.6
            },
            'min_spread': {
                'min': 0.1,
                'max': 1.0,
                'step': 0.05,
                'default': 0.25
            },
            'position_size': {
                'min': 1,
                'max': 5,
                'step': 1,
                'default': 1
            }
        }
    
    @classmethod
    def get_default_config(cls) -> Dict[str, Any]:
        """Provide default configuration"""
        return {
            'imbalance_threshold': 0.6,
            'min_spread': 0.25,
            'position_size': 1,
            'max_position': 3,
            'stop_loss_ticks': 4,
            'profit_target_ticks': 2
        }
```

### Bid-Ask Bounce Strategy

```python
@register_strategy(
    "bid_ask_bounce",
    description="Trades bid-ask bounce patterns",
    requires_l2_data=False
)
class BidAskBounceStrategy(BaseStrategy):
    """Strategy that trades bid-ask bounce patterns"""
    
    def __init__(self, config: Dict[str, Any]):
        super().__init__(config)
        self.bounce_threshold = config.get('bounce_threshold', 2)
        self.min_bounce_size = config.get('min_bounce_size', 0.5)
        self.hold_periods = config.get('hold_periods', 10)
        
        # Strategy state
        self.price_history = deque(maxlen=20)
        self.entry_price = None
        self.periods_held = 0
        
    def generate_signal(self, market_data: MarketData) -> Signal:
        """Detect and trade bounce patterns"""
        mid_price = (market_data.bid + market_data.ask) / 2
        self.price_history.append(mid_price)
        
        # Need history to detect patterns
        if len(self.price_history) < 5:
            return Signal('HOLD', 0)
        
        # Manage existing position
        if self.position != 0:
            return self._manage_position(market_data)
        
        # Look for bounce patterns
        bounce_signal = self._detect_bounce()
        if bounce_signal:
            return Signal(
                bounce_signal['action'],
                1,
                metadata={'pattern': bounce_signal['pattern']}
            )
        
        return Signal('HOLD', 0)
    
    def _detect_bounce(self) -> Optional[Dict[str, Any]]:
        """Detect bounce patterns in price history"""
        if len(self.price_history) < 5:
            return None
        
        recent_prices = list(self.price_history)[-5:]
        
        # Detect bid bounce (V pattern)
        if (recent_prices[2] < recent_prices[0] and 
            recent_prices[2] < recent_prices[4] and
            recent_prices[0] - recent_prices[2] >= self.min_bounce_size):
            
            bounce_count = self._count_bounces(recent_prices)
            if bounce_count >= self.bounce_threshold:
                return {
                    'action': 'BUY',
                    'pattern': 'bid_bounce',
                    'bounce_size': recent_prices[0] - recent_prices[2]
                }
        
        # Detect ask bounce (^ pattern)
        if (recent_prices[2] > recent_prices[0] and 
            recent_prices[2] > recent_prices[4] and
            recent_prices[2] - recent_prices[0] >= self.min_bounce_size):
            
            bounce_count = self._count_bounces(recent_prices)
            if bounce_count >= self.bounce_threshold:
                return {
                    'action': 'SELL',
                    'pattern': 'ask_bounce',
                    'bounce_size': recent_prices[2] - recent_prices[0]
                }
        
        return None
```

## Configuration Management

### Strategy Configuration System

```python
class StrategyConfigManager:
    """Manages strategy configurations"""
    
    def __init__(self, config_dir: str):
        self.config_dir = Path(config_dir)
        self.configs = {}
        self._load_configs()
    
    def get_config(self, strategy_name: str, profile: str = 'default') -> Dict[str, Any]:
        """Get configuration for strategy and profile"""
        key = f"{strategy_name}_{profile}"
        
        if key not in self.configs:
            # Try loading from file
            config_file = self.config_dir / f"{strategy_name}_{profile}.yaml"
            if config_file.exists():
                self.configs[key] = self._load_yaml(config_file)
            else:
                # Fall back to default
                self.configs[key] = strategy_registry.get_default_config(strategy_name)
        
        return self.configs[key].copy()
    
    def save_config(self, 
                   strategy_name: str, 
                   config: Dict[str, Any],
                   profile: str = 'default') -> None:
        """Save configuration to file"""
        config_file = self.config_dir / f"{strategy_name}_{profile}.yaml"
        self._save_yaml(config_file, config)
        
        # Update cache
        key = f"{strategy_name}_{profile}"
        self.configs[key] = config.copy()
    
    def validate_config(self, 
                       strategy_name: str, 
                       config: Dict[str, Any]) -> ValidationResult:
        """Validate configuration against strategy requirements"""
        try:
            # Try to instantiate strategy with config
            strategy = strategy_registry.get_strategy(strategy_name, config)
            
            # Check parameter bounds
            params = strategy.get_parameters()
            for param_name, param_spec in params.items():
                if param_name in config:
                    value = config[param_name]
                    if value < param_spec['min'] or value > param_spec['max']:
                        return ValidationResult(
                            False, 
                            f"Parameter {param_name} out of bounds"
                        )
            
            return ValidationResult(True, "Configuration valid")
            
        except Exception as e:
            return ValidationResult(False, str(e))
```
